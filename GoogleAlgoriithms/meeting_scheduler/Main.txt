"""
Bottom-up approach

1. identify and design the smaller components first (interval and a meeting room)
2. use small components to design additional and more complex components. (meeting and calendar)
3. repeat the steps util we design the scheduler

"""

from enum import Enum

    WAITING = 1
    IN_PROGRESS = 2
    FINISHED = 3

class Interval:

        self.start = start
        self.end = end

    def __init__(self, id, capacity, is_available):

        self.capacity = capacity
        self.is_available = is_available
        self.booked_intervals = []


class Calendar:
    def __init__(self):
        self.meetings = []

        pass
